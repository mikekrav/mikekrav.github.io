How to Add an RSS Feed from Medium

Below is sample code that shows how to fetch and display an RSS feed from a Medium blog. This uses JavaScript to fetch the feed, parse it, and display the articles on your page.

// JavaScript code to fetch and display Medium RSS feed
document.addEventListener('DOMContentLoaded', function() {
    const mediumRssFeed = 'https://medium.com/feed/@yourusername'; // Replace with your Medium username
    const rssToJsonApi = 'https://api.rss2json.com/v1/api.json?rss_url=';
    
    fetch(rssToJsonApi + encodeURIComponent(mediumRssFeed))
        .then(response => response.json())
        .then(data => {
            if (data.status === 'ok') {
                displayArticles(data.items);
            } else {
                document.getElementById('medium-articles').innerHTML = 
                    '<p>Error loading articles. Please check back later.</p>';
            }
        })
        .catch(error => {
            console.error('Error fetching Medium RSS feed:', error);
            document.getElementById('medium-articles').innerHTML = 
                '<p>Error loading articles. Please check back later.</p>';
        });
    
    function displayArticles(articles) {
        const container = document.getElementById('medium-articles');
        let html = '';
        
        articles.forEach(article => {
            // Extract first paragraph as excerpt
            const content = document.createElement('div');
            content.innerHTML = article.content;
            const paragraphs = content.querySelectorAll('p');
            let excerpt = '';
            
            if (paragraphs.length > 0) {
                excerpt = paragraphs[0].textContent.trim();
                if (excerpt.length > 150) {
                    excerpt = excerpt.substring(0, 147) + '...';
                }
            }
            
            // Format publication date
            const pubDate = new Date(article.pubDate);
            const formattedDate = pubDate.toLocaleDateString('en-US', {
                year: 'numeric', 
                month: 'long', 
                day: 'numeric'
            });
            
            html += `
                <div class="article">
                    <h3 class="article-title">
                        <a href="${article.link}" target="_blank">${article.title}</a>
                    </h3>
                    <div class="article-meta">${formattedDate}</div>
                    <p class="article-excerpt">${excerpt}</p>
                    <a href="${article.link}" class="article-link" target="_blank">
                        Read more
                    </a>
                </div>
            `;
        });
        
        container.innerHTML = html;
    }
});

To use this code, you'll need to:

  • Replace yourusername with your actual Medium username
  • Add a div with id="medium-articles" in your HTML where you want the articles to appear
  • The RSS-to-JSON API used in this example has a free tier that should be sufficient for most personal websites
